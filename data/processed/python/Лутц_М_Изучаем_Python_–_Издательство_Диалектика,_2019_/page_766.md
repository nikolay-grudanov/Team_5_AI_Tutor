#!/python
"""
Файл: formats.py (Python 2.x и 3.x)
Разнообразные специализированные функции для форматирования строк с целью отображения.
Тестирование можно производить с помощью заготовленного теста или аргументов командной строки.
Что сделать: добавить круглые скобки для отрицательных денежных сумм, реализовать больше возможностей.
"""
def commas(N):
    """
    Форматирует положительное целое число N для отображения с запятыми, разделяющими группы цифр: "xxx,yyy,zzz".
    """
    digits = str(N)
    assert digits.isdigit()
    result = ''
    while digits:
        digits, last3 = digits[:-3], digits[-3:]
        result = (last3 + ',' + result) if result else last3
    return result

def money(N, numwidth=0, currency='$'):
    """
    Форматирует число N для отображения с запятыми, 2 десятичными цифрами, ведущим символом $ и знаком, а также необязательным дополнением:
    "$ -xxx,yyy.zz".
    numwidth=0 - отсутствие дополнения пробелами,
    currency='' - опустить символ $
    или символ не ASCII для других валют (например, фунт - u'\xA3' или u'\u00A3').
    """
    sign = '-' if N < 0 else ''
    N = abs(N)
    whole = commas(int(N))
    fract = ('%.2f' % N)[-2:]
    number = '%s%s.%s' % (sign, whole, fract)
    return '%s%s%s' % (currency, numwidth, number)
