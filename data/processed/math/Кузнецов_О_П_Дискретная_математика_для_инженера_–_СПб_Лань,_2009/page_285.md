\[
x_1 i \Rightarrow x_1 0i;

\ldots \ldots \ldots
x_1 i \Rightarrow x_1 9i;
x_1 i \Rightarrow x_1 f;
x_1 f, x_2 f \Rightarrow (x_1 \vee x_2)f;
x_1 f, x_2 f \Rightarrow (x_1 & x_2)f;
x_1 f, x_2 f \Rightarrow (x_1 \rightarrow x_2)f;
